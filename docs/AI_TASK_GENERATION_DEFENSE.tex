\section{1. Giới thiệu}
%==========================================================================

\subsection{1.1 Motivation}
Trong quá trình phát triển phần mềm, việc chuyển đổi các yêu cầu (requirements) phức tạp thành các user stories nhỏ, rõ ràng là một thách thức lớn. Các vấn đề chính:

\begin{itemize}
    \item \textbf{Phân tích yêu cầu thủ công}: Mất thời gian, dễ bỏ sót chi tiết
    \item \textbf{Thiếu tính nhất quán}: Định dạng user story không đồng nhất
    \item \textbf{Không đánh giá INVEST}: Không kiểm tra tính hợp lệ của user story
    \item \textbf{Tính mở rộng kém}: Khó áp dụng cho các dự án lớn
\end{itemize}

\subsection{1.2 Giải pháp đề xuất}
Xây dựng một \textbf{AI Task Generation Module} tự động:
\begin{itemize}
    \item Phân tích yêu cầu bằng NLP (Natural Language Processing)
    \item Phân cấp yêu cầu theo múc độ phức tạp
    \item Tạo user stories theo format Agile chuẩn
    \item Kiểm tra INVEST compliance
    \item Phát hiện gaps và risks
\end{itemize}

\subsubsection{2.2.1 Requirement Parser}
\begin{itemize}
    \item \textbf{Input}: Tệp yêu cầu (TXT, MD, DOCX, PDF)
    \item \textbf{Xử lý}:
    \begin{enumerate}
        \item Extract text từ đa định dạng
        \item Tokenization + Sentence segmentation
        \item Named Entity Recognition (NER)
        \item Dependency parsing
    \end{enumerate}
    \item \textbf{Output}: Cấu trúc dữ liệu requirement có cấu trúc
\end{itemize}

\subsubsection{2.2.2 Semantic Analyzer}
\begin{itemize}
    \item Phân loại requirement theo loại:
    \begin{itemize}

        \item \textbf{Security}: Bảo mật (15\%)
        \item \textbf{Performance}: Hiệu năng (10\%)
        \item \textbf{Interface}: Giao diện (5\%)
    \end{itemize}
    \item Trích xuất domain, role, action
    \item Tính semantic similarity giữa requirements
\end{itemize}

\section{4. Các Thuật toán Chính}
%==========================================================================

\subsection{4.1 Semantic Similarity Matching}

Sử dụng TF-IDF + Cosine Similarity:

$$\text{similarity}(req_i, req_j) = \frac{\vec{v_i} \cdot \vec{v_j}}{|\vec{v_i}| \times |\vec{v_j}|}$$

\textbf{Ứng dụng}: Phát hiện duplicate requirements, grouping related requirements

\subsection{4.2 Priority Scoring}

$$P = w_1 \cdot \text{business\_value} + w_2 \cdot \text{complexity} + w_3 \cdot \text{risk}$$

Trong đó:
\begin{itemize}
    \item $w_1 = 0.5$ (Giá trị business)
    \item $w_2 = 0.3$ (Độ phức tạp)
    \item $w_3 = 0.2$ (Rủi ro)
\end{itemize}

\subsection{4.3 INVEST Score Calculation}

$$INVEST = \sqrt{\frac{\sum I_i}{N}}$$

Với automatic scoring cho mỗi tiêu chí (0.0 - 1.0)

\subsection{A. Ví dụ Decomposition}

\textbf{Input Requirement:}
\begin{verbatim}
"Phải quản lý hồ sơ bệnh nhân với tất cả thông tin cá nhân, 
lịch sử bệnh, kết quả xét nghiệm, và lịch sử điều trị"
\end{verbatim}

\textbf{Output User Stories (5 tasks):}

\begin{enumerate}
    \item \textbf{Happy Path}: Quản lý hồ sơ bệnh nhân - Happy Path
    \begin{itemize}
        \item As a Nurse, I want to create patient record, so that I can store patient info
        \item AC: When nurse clicks "Add Patient", form appears with required fields
    \end{itemize}
    
    \item \textbf{Edge Cases}: Quản lý hồ sơ bệnh nhân - Edge Cases \& Validation
    \begin{itemize}
        \item Validate duplicate patient ID
        \item Handle incomplete information
    \end{itemize}
    
    \item \textbf{CRUD Create}: Quản lý hồ sơ bệnh nhân - Tạo Mới
    \begin{itemize}
        \item Create new patient record in system
    \end{itemize}
    
    \item \textbf{CRUD Read}: Quản lý hồ sơ bệnh nhân - Xem/Tra Cứu
    \begin{itemize}
        \item View/Search patient records by ID or name
    \end{itemize}
    
    \item \textbf{CRUD Update}: Quản lý hồ sơ bệnh nhân - Cập Nhật
    \begin{itemize}
        \item Update patient information, history
    \end{itemize}
\end{enumerate}

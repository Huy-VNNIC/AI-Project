\subsection{Template 1: Khi đồng ý và làm theo}

\begin{quote}
\textbf{Template Response:}

\textit{We thank the reviewer for this excellent suggestion. We have [specific action taken]:}

\begin{itemize}
    \item \textit{[Specific change 1 with location in manuscript]}
    \item \textit{[Specific change 2]}
\end{itemize}

\textbf{Changes in manuscript:}
\begin{itemize}
    \item \textit{Section X: [description of change]}
    \item \textit{Figure/Table Y: [description]}
\end{itemize}

\textbf{Results:} \textit{[Outcome of the change, e.g., "Calibrated COCOMO II improved MMRE from 2.79 to 1.85, but RF still outperforms (0.65, p<0.001)."]}
\end{quote}

\subsection{Template 2: Khi giải thích limitation}

\begin{quote}
\textbf{Template Response:}

\textit{We appreciate this valuable comment. While we acknowledge [concern], we respectfully note that [limitation/constraint]:}

\begin{itemize}
    \item \textit{[Reason 1]}
    \item \textit{[Reason 2]}
\end{itemize}

\textit{However, we have [partial mitigation]:}
\begin{itemize}

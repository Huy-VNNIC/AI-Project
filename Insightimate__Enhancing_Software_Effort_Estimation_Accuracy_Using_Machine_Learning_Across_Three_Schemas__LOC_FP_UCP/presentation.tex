% ---- SLIDE 2: THE PROBLEM & CURRENT STATE  (1 min) ----
\begin{frame}{The Problem: Software Effort Estimation (SEE)}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Why It Matters:}
\begin{itemize}
    \item Chaos Report (Standish Group): only \textbf{29\%} of projects succeed
    \item \textbf{52\%} challenged (delayed/over-budget), \textbf{19\%} fail completely
    \item Root cause: \textcolor{darkred}{\textbf{inaccurate effort estimation}}
\end{itemize}

\vspace{0.3cm}
\textbf{Current Methods \& Limitations:}
\begin{itemize}
    \item \textbf{Parametric} (COCOMO~II): transparent but rigid on heterogeneous data
    \item \textbf{ML} (RF, XGBoost): flexible but \emph{lacks a reproducible, fair evaluation framework}
\end{itemize}

\column{0.5\textwidth}
\begin{block}{Three Sizing Schemas in Practice}
\begin{itemize}
    \item \textbf{LOC} (Lines of Code): 2,765 projects --- 90.5\%
    \item \textbf{FP} (Function Points): 158 projects --- 5.2\%
    \item \textbf{UCP} (Use Case Points): 131 projects --- 4.3\%
\end{itemize}
\end{block}

\vspace{0.25cm}
\begin{alertblock}{Core Problem}
LOC, FP, and UCP are studied in \textbf{isolation}. No unified, fair, and reproducible cross-schema benchmarking framework exists.
\end{alertblock}
\end{columns}
\end{frame}

% ---- SLIDE 3: RESEARCH GAPS  (1 min) ----
\begin{frame}{Research Gaps \& Research Questions}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{Three Unresolved Research Gaps:}

\vspace{0.25cm}
\begin{enumerate}
    \item \textcolor{darkred}{\textbf{(G1) Weak Dataset Auditability:}}\\
    Most studies cite ``public datasets'' without provenance, deduplication rules, or rebuild scripts.

    \vspace{0.2cm}
    \item \textcolor{darkred}{\textbf{(G2) Unfair Parametric Baselines:}}\\
    COCOMO~II applied with default parameters when cost drivers are missing $\Rightarrow$ \textit{straw-man comparisons}.

    \vspace{0.2cm}
    \item \textcolor{darkred}{\textbf{(G3) Ambiguous Aggregated Reporting:}}\\
    Micro-averaging dominated by LOC (90.5\%), masking true FP/UCP behavior.
\end{enumerate}

\column{0.45\textwidth}
\begin{exampleblock}{Research Questions}
\small\textbf{RQ1:} Do ensemble methods (RF, XGBoost) outperform a \emph{calibrated} baseline across LOC/FP/UCP?

\vspace{0.1cm}
\small\textbf{RQ2:} Does macro-averaging with LOSO provide a more honest evaluation than micro-averaging?
\end{exampleblock}

\vspace{0.15cm}
\begin{block}{Primary Contribution}
\small A \textbf{reproducible cross-schema benchmarking framework} for SEE --- a reusable methodological artifact for the community.
\end{block}
\end{columns}
\end{frame}

\column{0.48\textwidth}
\begin{block}{Schema-Specific Protocols (G3)}
\small\textbf{LOC} ($n$=2,765):
\begin{itemize}\small
    \item 80/20 $\times$ 10 seeds + \textbf{LOSO} (11-fold)
\end{itemize}
\textbf{FP} ($n$=158, small sample):
\begin{itemize}\small
    \item \textbf{LOOCV} (158-fold) + Bootstrap CI
\end{itemize}
\textbf{UCP} ($n$=131):
\begin{itemize}\small
    \item 80/20 $\times$ 10 seeds + 5-fold CV
\end{itemize}
\end{block}

\begin{alertblock}{Contribution \#3}
\small Schema-appropriate evaluation protocols.
\end{alertblock}
\end{columns}

% ---- SLIDE 6: TWO METHODOLOGICAL INNOVATIONS  (1 min) ----
\begin{frame}{Two Core Methodological Innovations}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Innovation \#1 --- Calibrated Baseline}\\{\small (addressing G2)}
\vspace{0.05cm}

Instead of COCOMO~II with default parameters, fit a power-law model \textit{on training data}:
\begin{equation}
\log(E) = \alpha + \beta \cdot \log(\text{Size})
\end{equation}
\small $(\alpha, \beta)$ via \texttt{curve\_fit} (Levenberg-Marquardt), per-schema, per-seed.

\vspace{0.1cm}
$\Rightarrow$ \textbf{Fair lower bound:} ML must beat an already-optimized parametric model.

\vspace{0.1cm}
\textbf{Innovation \#3 --- Tail Evaluation}\\{\small Separate analysis of top 10\% highest-effort projects.}

\column{0.5\textwidth}
\textbf{Innovation \#2 --- Macro-Averaging}\\{\small (addressing G3)}
\vspace{0.05cm}

\begin{block}{Problem: Micro-averaging}
\small LOC = 90.5\% $\Rightarrow$ overall metric dominated by LOC.
\end{block}

\begin{exampleblock}{Solution: Macro-averaging}
\small
\vspace{-0.1cm}
\begin{equation}
m_{\text{macro}} = \frac{1}{3}(m_{\text{LOC}} + m_{\text{FP}} + m_{\text{UCP}})
\end{equation}
\vspace{-0.2cm}
Each schema contributes \textbf{33.3\%} regardless of $n$.
\end{exampleblock}

\vspace{0.1cm}
\begin{alertblock}{Combined Effect}
\small All three innovations together address G1--G3 and ensure fair, reproducible comparison.
\end{alertblock}
\end{columns}
\end{frame}

\begin{frame}{[Backup] Anticipated Q\&A}
\footnotesize
\begin{enumerate}
    \item \textbf{Why Random Forest over Deep Learning?}\\
    DL requires more data (FP: $n$=158 only). RF is interpretable and stable on small tabular datasets.

    \item \textbf{How is overfitting controlled?}\\
    LOOCV (FP), LOSO CV, 10 seeds, OOB score, IQR capping.

    \item \textbf{Why report MMRE despite its bias?}\\
    Supplementary metric for literature comparison; \textbf{MAE is primary}. Standard SEE practice (Shepperd \& Kadoda, 2001).

    \item \textbf{Is dataset deduplication sufficient?}\\
    Deduplication by project ID, effort range, and year. Residual risk noted in Threats to Validity.

    \item \textbf{Why not use full COCOMO~II as baseline?}\\
    Most FP/UCP datasets lack cost drivers --- default parameters create straw-man comparisons (gap G2).
\end{enumerate}
\end{frame}

    COCOMO~II dùng tham số mặc định khi thiếu cost drivers $\Rightarrow$ so sánh kiểu \textit{straw-man} (người rơm).

    \vspace{0.2cm}
    \item \textcolor{darkred}{\textbf{(G3) Tổng hợp kết quả mơ hồ:}}\\
    Micro-averaging bị chi phối bởi LOC (90.5\%), che khuất hiệu năng thực của FP/UCP.
\end{enumerate}

\begin{exampleblock}{Câu hỏi nghiên cứu}

\vspace{0.2cm}

\end{exampleblock}

\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Đổi mới \#1 — Calibrated Baseline (giải quyết G2)}

\vspace{0.1cm}
Thay vì COCOMO~II với tham số mặc định, hiệu chỉnh power-law \textit{trên tập huấn luyện}:
\begin{equation}
\log(E) = \alpha + \beta \cdot \log(\text{Size})
\end{equation}
$(\alpha, \beta)$ fit bằng \texttt{curve\_fit} (Levenberg-Marquardt), per-schema, per-seed.

\vspace{0.2cm}
$\Rightarrow$ \textbf{Lower bound công bằng:} ML phải vượt baseline đã được tối ưu hóa.

\column{0.5\textwidth}
\textbf{Đổi mới \#2 — Macro-Averaging (giải quyết G3)}

\begin{block}{Vấn đề: Micro-averaging}
LOC chiếm 90.5\% $\Rightarrow$ kết quả tổng bị chi phối bởi LOC.
\end{block}

\begin{exampleblock}{Giải pháp: Macro-averaging}
\begin{equation}
m_{\text{macro}} = \frac{1}{3}(m_{\text{LOC}} + m_{\text{FP}} + m_{\text{UCP}})
\end{equation}
Mỗi lược đồ đóng góp \textbf{33.3\%} bất kể $n$.
\end{exampleblock}

\vspace{0.15cm}
\textbf{Đổi mới \#3 — Tail Evaluation}\\
Phân tích riêng top 10\% dự án lớn nhất (long-tail distribution).
\end{columns}

\begin{frame}{[Backup] Câu hỏi Hội đồng thường gặp}
\small
\begin{enumerate}
    \item \textbf{Tại sao chọn RF thay vì Deep Learning?}\\
    DL đòi hỏi nhiều dữ liệu hơn (FP chỉ $n$=158). RF cho kết quả tốt, có thể giải thích được (feature importance), và ổn định hơn trên dữ liệu bảng nhỏ.

    \vspace{0.15cm}
    \item \textbf{Overfitting được kiểm soát như thế nào?}\\
    (a) LOOCV cho FP; (b) LOSO cross-validation; (c) 10 independent seeds; (d) OOB score monitoring; (e) IQR capping tránh outlier ảnh hưởng.

    \vspace{0.15cm}
    \item \textbf{Tại sao dùng MMRE dù biết nó có bias?}\\
    MMRE là chỉ số \textit{bổ sung} để so sánh với tài liệu; \textbf{MAE là chỉ số chính}. Đây là thông lệ chuẩn trong SEE (Shepperd \& Kadoda, 2001).

    \vspace{0.15cm}
    \item \textbf{Dataset có thể bị trùng lặp không?}\\
    Đã áp dụng cross-source deduplication theo project ID, effort range và năm. Rủi ro còn lại được ghi nhận trong Threats to Validity.

    \vspace{0.15cm}
    \item \textbf{Tại sao không dùng COCOMO~II đầy đủ làm baseline?}\\
    Hầu hết tập FP/UCP không có cost drivers. Dùng COCOMO~II với tham số mặc định tạo \textit{straw-man comparison} --- đây chính là khoảng trống G2 mà nghiên cứu này giải quyết.
\end{enumerate}

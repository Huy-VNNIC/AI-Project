\section{Current Problem}

Modern software testing process:
\begin{itemize}
    \item \textbf{Time}: 30-60 minutes per requirement
    \item \textbf{Cost}: \$50-150 per hour × 40-60 hours = \$2,000-9,000 per project
    \item \textbf{Accuracy}: 70-80\% (easy to miss edge cases)
    \item \textbf{Reusability}: 0\% - each project starts from scratch
\end{itemize}

\textcolor{red}{\textbf{Problem: Repetitive work, time-consuming, expensive, unstable quality}}

\section{Proposed Solution}

Build an AI system that automatically generates test cases:

\begin{equation}
\text{Speed} = \frac{30 \text{ minutes}}{50 \text{ ms}} = 36,000\times \text{ faster}
\end{equation}

\begin{equation}
\text{Cost} = \frac{\$10,000}{\$200} = 97-98\% \text{ savings}
\end{equation}

\begin{equation}
\text{Accuracy} = 85-95\% \text{ depending on mode}
\end{equation}

\section{Key Objectives}

\begin{enumerate}
    \item Automatically generate test cases from requirements (NLP)
    \item Classify 7 different test types (Functional, Edge, Security, Performance, Integration, Regression, Threat)
    \item Integrate threat modeling (OWASP)
    \item Validate with 9 real-world production systems
    \item 100\% pass rate on test suite
\end{enumerate}

\section{Processing Pipeline}

\begin{verbatim}
INPUT: Requirement text
    |
[Step 1] NLP analysis (spaCy)
    |
[Step 2] Keyword extraction
    |
[Step 3] 7-category classification
    - Functional: Basic workflows
    - Edge cases: Boundary conditions
    - Security: Vulnerabilities (8 OWASP)
    - Performance: Latency/throughput
    - Integration: Component interaction
    - Regression: Previous fixes
    - Threat: Real-world attacks
    |
[Step 4] Deduplication (remove 15-25%)
    |
[Step 5] Confidence scoring (0-100%)
    |
OUTPUT: Test cases with risk levels
\end{verbatim}

\section{Group 2: Technology and Architecture}

\subsection{Q4: What technologies does the system use?}

\textbf{Answer}:
\begin{itemize}

\subsection{Q5: Why spaCy and not BERT or GPT?}

\textbf{Answer}:
\begin{itemize}
    \item \textbf{spaCy}: Fast (50ms), explainable, low resource
    \item \textbf{BERT/GPT}: Slow (500ms+), expensive GPU, hard to control
    \item \textbf{Future}: Phase 2 integrates BERT/GPT when usage scales
    \item \textbf{Now}: Rule-based + spaCy sufficient for capstone
\end{itemize}

\subsection{Q6: What are the 3 modes?}

\textbf{Answer}:
\begin{itemize}
    \item \textbf{Rule-Based}: Pure keyword + patterns, 85\% accuracy, 20ms
    \item \textbf{Hybrid}: Rule + simple ML, 90\% accuracy, 25ms
    \item \textbf{Transformer}: Full spaCy parsing, 95\% accuracy, 100ms
    \item Users choose based on latency/accuracy tradeoff
\end{itemize}

\subsection{Q7: How do you prevent duplicate tests?}

\textbf{Answer}:
\begin{itemize}
    \item \textbf{String similarity}: Levenshtein distance
    \item \textbf{Semantic similarity}: spaCy word vectors
    \item \textbf{Category dedup}: No 2 identical "Functional" tests
    \item \textbf{Result}: 15-25\% duplicate removal, keep 75-85\% unique
\end{itemize}

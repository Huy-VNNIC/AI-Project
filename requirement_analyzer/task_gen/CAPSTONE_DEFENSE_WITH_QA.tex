\section{Nhóm 1: Vấn Đề & Giải Pháp}

\subsection{Q1: Vấn đề hiện tại trong testing là gì?}

\textbf{Trả lời}: 
\begin{itemize}
    \item Manual testing tốn 30-60 phút mỗi requirement
    \item 40-60 requirements = 1,500 giờ human effort
    \item Chi phí: \$50-150/giờ = \$75,000-225,000 mỗi project
    \item Lỗi do con người: 20-30\% cases bị bỏ sót
    \item Không tái sử dụng được patterns
\end{itemize}

\subsection{Q2: Tại sao bạn chọn cách tiếp cận AI, không phải template?}

\textbf{Trả lời}:
\begin{itemize}
    \item Template là static - áp dụng cho mọi requirement như nhau
    \item AI hiểu ngữ cảnh - extract key concepts từ text
    \item AI tự động phát hiện edge cases, threats
    \item Example: "Login with email" → AI sinh 11 test != Template sinh 3-5 test
    \item AI học từ kiểu dữ liệu, độ phức tạp
\end{itemize}

\subsection{Q3: Độ chính xác 85-95\% có đủ tốt không?}

\textbf{Trả lời}:
\begin{itemize}
    \item Manual: 70-80\% (dễ bỏ sót edge cases)
    \item AI: 85-95\% (code coverage tốt + threat modeling)
    \item Lợi ích: Dù 10-15\% lower accuracy, nhưng FULL COVERAGE thay vì miss 50\%
    \item Engineer có thể review & refine - không phải viết từ 0
    \item Cost benefit: 97\% tiết kiệm > Perfect accuracy
\end{itemize}

\section{Nhóm 2: Công Nghệ & Kiến Trúc}

\subsection{Q4: Hệ thống dùng những công nghệ gì?}

\textbf{Trả lời}:
\begin{itemize}
    \item \textbf{NLP}: spaCy (tokenization, POS tagging, dependency parsing)
    \item \textbf{Rule-based**: Keywords + pattern matching (high precision)
    \item \textbf{Backend}: Python 3.10 + FastAPI
    \item \textbf{Testing}: pytest (8 comprehensive test suites)
    \item \textbf{Deployment}: Docker + Docker Compose
\end{itemize}

\subsection{Q5: Sao bạn dùng spaCy chứ không dùng BERT, GPT?}

\textbf{Trả lời}:
\begin{itemize}
    \item \textbf{spaCy}: Fast (50ms), explainable, low resource
    \item \textbf{BERT/GPT}: Slow (500ms+), expensive GPU, hard to control
    \item \textbf{Tương lai}: Phase 2 có integrate BERT/GPT khi tần suất dùng tăng
    \item \textbf{Hiện tại}: Rule-based + spaCy đủ precision cho capstone
\end{itemize}

\subsection{Q6: 3 chế độ (Rule, Hybrid, Transformer) là gì?}

\textbf{Trả lời}:
\begin{itemize}
    \item \textbf{Rule-Based}: Pure keyword + patterns, 85\% accuracy, 20ms
    \item \textbf{Hybrid}: Rule + simple ML scoring, 90\% accuracy, 25ms
    \item \textbf{Transformer}: Full spaCy dependency parsing, 95\% accuracy, 100ms
    \item Người dùng chọn mode tuỳ theo latency/accuracy tradeoff
\end{itemize}

\subsection{Q7: Làm sao bạn tránh duplicate test cases?}

\textbf{Trả lời}:
\begin{itemize}
    \item \textbf{String similarity}: Levenshtein distance
    \item \textbf{Semantic similarity}: spaCy word vectors
    \item \textbf{Category deduplication}: Không sinh 2 "Functional" tests giống nhau
    \item \textbf{Result}: 15-25\% duplicate removal, giữ 75-85\% unique tests
\end{itemize}
